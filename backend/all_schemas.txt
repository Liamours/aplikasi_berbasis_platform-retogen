from pydantic import BaseModel
from typing import Literal

class AddArticle(BaseModel):
    token: str
    article_title: str
    article_preview: str
    article_content: str
    article_tag: Literal["office", "budget", "gaming", "flagship"]
    article_image: str

    token: str
    article_id: str

    article_preview: Optional[str] = None
    article_content: Optional[str] = None
    article_tag: Optional[Literal["office", "budget", "gaming", "flagship"]] = None
    article_image: Optional[str] = None  # base64
